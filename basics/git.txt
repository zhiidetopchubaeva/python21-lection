git init   - создаем git репозиторий из текущей папки
git remote add название(или origin) ssh-ссылка - сохраняем ssh - ссылку в переменную
git remote -v - посмотреть все сохраненные  ssh-ссылки
git add .   - просто пишем эту комманду каждый раз перед commit
git commit -m 'message'   - создаем новую версию
git push origin master   - закидываем версию в github
git pull origin master - стягиваем новые изменения с githuba  (если их нет в локальном, но есть в github)
git clone название файла - файл клонируем чужой
git log - посмотреть все коммиты

git branch - показывает все ветки (* показывает ветку, где мы находимся)
git branch название - создать новую ветку (мы просто создаем ветку,но не переходим)
git branch -D название ветки -  удалить ветку
git checkout название ветки - перейти на ветку 
git checkout -b  название  - создать ветку и перейти на нее


!!!!!!!! git reset HEAD~     - отменяет последний коммит (как git commit, но наоборот)
git restore .   - работает также как add, но наоборот
